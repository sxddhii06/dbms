1) Display employee information

SELECT Empno, Ename, Job, Mgr, Joined_date,
       Salary, Commission, Deptno, Address
FROM Employee;

2) Display unique jobs

SELECT DISTINCT Job
FROM Employee;

3) Change location of dept 40 to Bangalore

UPDATE Department
SET Location = 'Bangalore'
WHERE Deptno = 40;

4) Change employee 1003 name to Nikhil Gosavi

UPDATE Employee
SET Ename = 'Nikhil Gosavi'
WHERE Empno = 1003;

5) Delete Pushkar Deshpande from employee table

DELETE FROM Employee
WHERE Ename = 'Pushkar Deshpande';

6) Display employees whose job is Manager or Analyst
---Using OR operator---

SELECT *
FROM Employee
WHERE Job = 'Manager'
   OR Job = 'Analyst';

---Using IN operator---

SELECT *
FROM Employee
WHERE Job IN ('Manager', 'Analyst');

7) Display employee name and department number
of employees in dept 10,20,30,40

SELECT Ename, Deptno
FROM Employee
WHERE Deptno IN (10,20,30,40);

8) Find names and joined date of employees
whose names start with 'A'

SELECT Ename, Joined_date
FROM Employee
WHERE Ename LIKE 'A%';

9) Find employee names having 'i'
as second letter

SELECT Ename
FROM Employee
WHERE Ename LIKE '_i%';

10) Find department number and maximum salary
where maximum salary is greater than 5000

SELECT Deptno, MAX(Salary) AS Maximum_Salary
FROM Employee
GROUP BY Deptno
HAVING MAX(Salary) > 5000;
