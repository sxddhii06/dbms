Problem Statement 1
Design at least 10 SQL queries for suitable database application using SQL DDL
statements
Consider the following database where the primary keys are underlined. Create the
following tables in oracle
Employee (employee_name,street, city)
Works (employee_name,company_name,salary)
Company (company_name,city)
Manages (employee_name,manager_name)



1) Create all four tables with appropriate constraints

Employee Table 
CREATE TABLE Employee (
    employee_name VARCHAR2(50) PRIMARY KEY,
    street        VARCHAR2(100) NOT NULL,
    city          VARCHAR2(50) NOT NULL
);

Company Table 
CREATE TABLE Company (
    company_name VARCHAR2(50) PRIMARY KEY,
    city         VARCHAR2(50) NOT NULL
);

Works Table 
CREATE TABLE Works (
    employee_name VARCHAR2(50),
    company_name  VARCHAR2(50),
    salary        NUMBER(8,2),

    CONSTRAINT pk_works PRIMARY KEY(employee_name, company_name),

    CONSTRAINT fk_works_employee
        FOREIGN KEY(employee_name)
        REFERENCES Employee(employee_name),

    CONSTRAINT fk_works_company
        FOREIGN KEY(company_name)
        REFERENCES Company(company_name)
);

Manages Table 
CREATE TABLE Manages (
    employee_name VARCHAR2(50),
    manager_name  VARCHAR2(50),

    CONSTRAINT pk_manages PRIMARY KEY(employee_name),

    CONSTRAINT fk_manages_employee
        FOREIGN KEY(employee_name)
        REFERENCES Employee(employee_name),

    CONSTRAINT fk_manages_manager
        FOREIGN KEY(manager_name)
        REFERENCES Employee(employee_name)
);
